-- =============================================================================
-- MIGRATION: RPC soft_delete_page_cascade - cascade soft delete de pages
-- Date: 2026-05-23
-- Description:
--   Marca page + todos descendentes (subpages) + databases ancorados
--   (boards.page_id IN descendants) como state='deleted'.
--   Usado pelo UI ao deletar page no sidebar.
-- =============================================================================

CREATE OR REPLACE FUNCTION public.soft_delete_page_cascade(_page_id UUID)
RETURNS INTEGER
LANGUAGE plpgsql SECURITY DEFINER SET search_path = public
AS $$
DECLARE
  affected_count INTEGER := 0;
BEGIN
  -- 1. Coletar IDs recursivos de pages (page + todas descendentes)
  WITH RECURSIVE page_tree AS (
    SELECT id FROM public.pages WHERE id = _page_id
    UNION ALL
    SELECT p.id FROM public.pages p
    INNER JOIN page_tree pt ON p.parent_id = pt.id
    WHERE p.state = 'active'
  )
  -- 2. Soft delete em pages
  UPDATE public.pages
  SET state = 'deleted'
  WHERE id IN (SELECT id FROM page_tree);
  GET DIAGNOSTICS affected_count = ROW_COUNT;

  -- 3. Soft delete em boards (databases) ancorados em qualquer page descendente
  WITH RECURSIVE page_tree AS (
    SELECT id FROM public.pages WHERE id = _page_id
    UNION ALL
    SELECT p.id FROM public.pages p
    INNER JOIN page_tree pt ON p.parent_id = pt.id
  )
  UPDATE public.boards
  SET state = 'deleted'
  WHERE page_id IN (SELECT id FROM page_tree)
    AND state = 'active';

  RETURN affected_count;
END;
$$;

-- RLS: caller deve ter is_page_admin (mesma regra do DELETE direto)
-- SECURITY DEFINER significa que a checagem fica no client/RLS antes da chamada
-- Aqui adicionamos uma checagem inline pra defesa em profundidade:
CREATE OR REPLACE FUNCTION public.soft_delete_page_cascade(_page_id UUID)
RETURNS INTEGER
LANGUAGE plpgsql SECURITY DEFINER SET search_path = public
AS $$
DECLARE
  affected_count INTEGER := 0;
  caller_id UUID := auth.uid();
BEGIN
  -- Defesa em profundidade: so admin da page pode cascade delete
  IF NOT public.is_page_admin(caller_id, _page_id) THEN
    RAISE EXCEPTION 'Permission denied: requires page admin';
  END IF;

  WITH RECURSIVE page_tree AS (
    SELECT id FROM public.pages WHERE id = _page_id
    UNION ALL
    SELECT p.id FROM public.pages p
    INNER JOIN page_tree pt ON p.parent_id = pt.id
    WHERE p.state = 'active'
  )
  UPDATE public.pages
  SET state = 'deleted'
  WHERE id IN (SELECT id FROM page_tree);
  GET DIAGNOSTICS affected_count = ROW_COUNT;

  WITH RECURSIVE page_tree AS (
    SELECT id FROM public.pages WHERE id = _page_id
    UNION ALL
    SELECT p.id FROM public.pages p
    INNER JOIN page_tree pt ON p.parent_id = pt.id
  )
  UPDATE public.boards
  SET state = 'deleted'
  WHERE page_id IN (SELECT id FROM page_tree)
    AND state = 'active';

  RETURN affected_count;
END;
$$;

import { nextKeyAfter } from '@/utils/lexorank';
import { supabase } from '@/integrations/supabase/client';

export const useCreatePage = () => {
  const qc = useQueryClient();
  return useMutation({
    retry: 1,
    mutationFn: async ({
      workspaceId,
      title,
      icon,
      parentId,
    }: {
      workspaceId: string;
      title: string;
      icon?: string;
      parentId?: string | null;
    }) => {
      const { data: user } = await supabase.auth.getUser();
      if (!user.user) throw new Error('Not authenticated');

      // Buscar ultimo sibling no nivel pra gerar sort_order
      let q = supabase
        .from('pages')
        .select('sort_order')
        .eq('workspace_id', workspaceId)
        .eq('state', 'active');
      q = parentId === null || parentId === undefined ? q.is('parent_id', null) : q.eq('parent_id', parentId);
      const { data: siblings } = await q.order('sort_order', { ascending: false }).limit(1);
      const lastSortOrder = siblings?.[0]?.sort_order ?? null;
      const newSortOrder = nextKeyAfter(lastSortOrder);

      // Manter `position` (campo retrocompat) com maxPosition+1000 igual antes
      const { data: lastPos } = await supabase
        .from('pages')
        .select('position')
        .eq('workspace_id', workspaceId)
        .eq('state', 'active')
        .order('position', { ascending: false })
        .limit(1);
      const maxPosition = lastPos?.[0]?.position ?? 0;

      const { data, error } = await supabase
        .from('pages')
        .insert({
          workspace_id: workspaceId,
          title,
          icon: icon ?? null,
          content: [],
          position: maxPosition + 1000,
          sort_order: newSortOrder,
          parent_id: parentId ?? null,
          created_by: user.user.id,
        })
        .select()
        .single();
      if (error) throw error;
      return data;
    },
    onSuccess: (_data, vars) => {
      qc.invalidateQueries({ queryKey: ['pages'] });
      qc.invalidateQueries({ queryKey: ['all-pages'] });
      qc.invalidateQueries({ queryKey: ['pages-tree', vars.workspaceId, vars.parentId ?? null] });
      // Se subpage, invalida child_count do pai (que mora na query do nivel ACIMA)
      // Como nao sabemos o nivel acima sem recursao, invalida toda a tree do workspace
      qc.invalidateQueries({ queryKey: ['pages-tree', vars.workspaceId] });
    },
  });
};

```typescript
import React from 'react';
import { Link } from 'react-router-dom';
import { ChevronRight } from 'lucide-react';
import { useQuery } from '@tanstack/react-query';
import { supabase } from '@/integrations/supabase/client';
import { useApp } from '@/context/AppContext';

interface PageBreadcrumbProps {
  pageId: string;
}

interface BreadcrumbNode {
  id: string;
  title: string;
}

/**
 * Renderiza trilha hierarquica da page: Workspace > Page raiz > ... > pageId atual.
 *
 * Estrategia de fetch: sobe a cadeia parent_id ate chegar em parent_id IS NULL.
 * Query unica com CTE recursivo via RPC seria mais eficiente, mas para hierarquias
 * tipicas (<5 niveis) loop client-side e suficiente.
 *
 * Cache: queryKey ['page-breadcrumb', pageId] com staleTime 60s.
 */
const PageBreadcrumb: React.FC<PageBreadcrumbProps> = ({ pageId }) => {
  const { workspaces } = useApp();

  const { data: chain = [] } = useQuery({
    queryKey: ['page-breadcrumb', pageId],
    enabled: !!pageId,
    staleTime: 60_000,
    queryFn: async () => {
      const result: BreadcrumbNode[] = [];
      let currentId: string | null = pageId;
      let workspaceId: string | null = null;
      // Limite defensivo: 20 niveis (evita loop infinito em caso de dado corrompido)
      for (let i = 0; i < 20 && currentId; i++) {
        const { data, error } = await supabase
          .from('pages')
          .select('id, title, parent_id, workspace_id')
          .eq('id', currentId)
          .single();
        if (error || !data) break;
        result.unshift({ id: data.id, title: data.title });
        workspaceId = data.workspace_id;
        currentId = (data as { parent_id: string | null }).parent_id;
      }
      return { chain: result, workspaceId };
    },
    select: (raw) => (raw as { chain: BreadcrumbNode[]; workspaceId: string | null }).chain,
  });

  const workspace = workspaces?.find((w: { id: string; name: string }) =>
    // Note: este useApp ja traz workspaces; encontramos pelo primeiro item da cadeia
    w.id === (chain[0] ? undefined : undefined) // placeholder; we fetch workspaceId em queryFn
  );

  if (chain.length === 0) return null;

  return (
    <nav className="flex items-center gap-1 text-xs text-muted-foreground mb-2" aria-label="Breadcrumb">
      {workspace && (
        <>
          <Link to={`/workspace/${workspace.id}`} className="hover:text-foreground transition-colors">
            {workspace.name}
          </Link>
          <ChevronRight className="w-3 h-3" />
        </>
      )}
      {chain.map((node, idx) => (
        <React.Fragment key={node.id}>
          {idx > 0 && <ChevronRight className="w-3 h-3" />}
          {idx === chain.length - 1 ? (
            <span className="text-foreground font-medium truncate max-w-[200px]" title={node.title}>{node.title}</span>
          ) : (
            <Link to={`/page/${node.id}`} className="hover:text-foreground transition-colors truncate max-w-[150px]" title={node.title}>
              {node.title}
            </Link>
          )}
        </React.Fragment>
      ))}
    </nav>
  );
};

export default PageBreadcrumb;
```

      // erro toasted pela mutation
