```sql
-- =============================================================================
-- MIGRATION: Database permissoes herdam page pai
-- Date: 2026-05-23
-- Description:
--   Atualiza RPC can_access_board para que quando boards.page_id IS NOT NULL
--   (database inline), delega checagem pra can_access_page (heranca da page).
--   Todas as RLS de items, columns, groups, board_views, column_values usam
--   can_access_board, entao a heranca propaga automaticamente.
-- Retrocompat: Boards sem page_id mantem comportamento atual.
-- =============================================================================

CREATE OR REPLACE FUNCTION public.can_access_board(_user_id UUID, _board_id UUID)
RETURNS BOOLEAN
LANGUAGE sql STABLE SECURITY DEFINER SET search_path = public
AS $$
  -- Branch 1: board e database inline (page_id IS NOT NULL) -> delega pra can_access_page
  SELECT
    CASE
      WHEN EXISTS (SELECT 1 FROM public.boards b WHERE b.id = _board_id AND b.page_id IS NOT NULL)
      THEN public.can_access_page(_user_id, (SELECT page_id FROM public.boards WHERE id = _board_id))
      ELSE (
        -- Branch 2: board tradicional, logica original
        EXISTS (
          SELECT 1 FROM public.boards b
          JOIN public.workspace_members wm ON wm.workspace_id = b.workspace_id
          WHERE b.id = _board_id AND wm.user_id = _user_id
        )
        OR EXISTS (
          SELECT 1 FROM public.board_permissions bp
          WHERE bp.board_id = _board_id AND bp.user_id = _user_id
        )
        OR EXISTS (
          SELECT 1 FROM public.user_roles ur
          WHERE ur.user_id = _user_id AND ur.role = 'admin'
        )
      )
    END
$$;

COMMENT ON FUNCTION public.can_access_board(UUID, UUID) IS
  'Quando board.page_id NOT NULL, delega pra can_access_page (database inline herda permissoes). Senao usa logica original (workspace_members | board_permissions | global admin).';
```
