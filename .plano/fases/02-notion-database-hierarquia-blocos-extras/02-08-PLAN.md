import React, { useState } from 'react';
import { useBoardViews } from '@/hooks/useBoardViews';
import { Table2, Kanban, Calendar, List, Plus } from 'lucide-react';
import CreateDatabaseViewDialog from './CreateDatabaseViewDialog';
import type { DatabaseViewType } from '@/types/database';

interface Props {
  boardId: string;
  activeViewId: string | null;
  onChangeView: (viewId: string) => void;
}

const ICON_BY_TYPE: Record<DatabaseViewType, React.ComponentType<{ className?: string }>> = {
  table: Table2,
  kanban: Kanban,
  calendar: Calendar,
  list_detailed: List,
};

const DatabaseViewTabs: React.FC<Props> = ({ boardId, activeViewId, onChangeView }) => {
  const { data: views = [] } = useBoardViews(boardId);
  const [createOpen, setCreateOpen] = useState(false);

  return (
    <div className="flex items-center gap-1 px-2 py-1 border-b border-border bg-muted/10 overflow-x-auto scrollbar-thin">
      {views.map((v) => {
        const type = v.view_type as DatabaseViewType;
        const Icon = ICON_BY_TYPE[type] ?? List;
        const isActive = v.id === activeViewId;
        return (
          <button
            key={v.id}
            onClick={() => onChangeView(v.id)}
            className={`flex items-center gap-1.5 px-2.5 py-1 rounded-md text-xs font-medium transition-colors whitespace-nowrap ${
              isActive
                ? 'bg-primary/15 text-primary'
                : 'text-muted-foreground hover:bg-muted hover:text-foreground'
            }`}
          >
            <Icon className="w-3.5 h-3.5" />
            <span className="truncate max-w-[120px]">{v.name}</span>
          </button>
        );
      })}
      <button
        onClick={() => setCreateOpen(true)}

```typescript
import React, { useState } from 'react';
import { Dialog, DialogContent, DialogHeader, DialogTitle, DialogFooter } from '@/components/ui/dialog';
import { Button } from '@/components/ui/button';
import { Input } from '@/components/ui/input';
import { Table2, Kanban, Calendar, List } from 'lucide-react';
import { useCreateBoardView } from '@/hooks/useBoardViews';
import { DATABASE_VIEW_TYPES, DATABASE_VIEW_LABELS, type DatabaseViewType } from '@/types/database';
import { toast } from 'sonner';

interface Props {
  open: boolean;
  onOpenChange: (open: boolean) => void;
  boardId: string;
  onCreated: (viewId: string) => void;
}

const ICONS: Record<DatabaseViewType, React.ComponentType<{ className?: string }>> = {
  table: Table2, kanban: Kanban, calendar: Calendar, list_detailed: List,
};

const CreateDatabaseViewDialog: React.FC<Props> = ({ open, onOpenChange, boardId, onCreated }) => {
  const [name, setName] = useState('');
  const [type, setType] = useState<DatabaseViewType>('table');
  const [isCreating, setIsCreating] = useState(false);
  const createView = useCreateBoardView();

  const handleConfirm = async () => {
    const trimmed = name.trim();
    if (!trimmed) { toast.error('Informe um nome'); return; }
    setIsCreating(true);
    try {
      const created = await createView.mutateAsync({
        boardId,
        name: trimmed,
        viewType: type,
        config: {},
      });
      onCreated(created.id);
      toast.success('View criada');
      setName('');
      setType('table');
      onOpenChange(false);
    } catch {
      toast.error('Erro ao criar view');
    } finally {
      setIsCreating(false);
    }
  };

  return (
    <Dialog open={open} onOpenChange={onOpenChange}>
      <DialogContent className="max-w-md">
        <DialogHeader>
          <DialogTitle>Nova view</DialogTitle>
        </DialogHeader>
        <div className="space-y-3">
          <Input
            value={name}
            onChange={(e) => setName(e.target.value)}
            placeholder="Nome da view"
            autoFocus
          />
          <div>
            <p className="text-xs font-medium text-muted-foreground mb-2">Tipo</p>
            <div className="grid grid-cols-2 gap-2">
              {DATABASE_VIEW_TYPES.map((t) => {
                const Icon = ICONS[t];
                const active = type === t;
                return (
                  <button
                    key={t}
                    onClick={() => setType(t)}
                    className={`flex items-center gap-2 px-3 py-2 rounded-md border text-sm transition-colors ${
                      active
                        ? 'border-primary bg-primary/10 text-primary'
                        : 'border-border hover:bg-muted'
                    }`}
                  >
                    <Icon className="w-4 h-4" />
                    {DATABASE_VIEW_LABELS[t]}
                  </button>
                );
              })}
            </div>
          </div>
        </div>
        <DialogFooter>
          <Button variant="outline" onClick={() => onOpenChange(false)} disabled={isCreating}>
            Cancelar
          </Button>
          <Button onClick={handleConfirm} disabled={isCreating}>
            {isCreating ? 'Criando...' : 'Criar view'}
          </Button>
        </DialogFooter>
      </DialogContent>
    </Dialog>
  );
};

export default CreateDatabaseViewDialog;
```

```typescript
import { useState, useEffect } from 'react';
import { useBoardViews } from '@/hooks/useBoardViews';

const DatabaseBlockView: React.FC<{ boardId: string; snapshotName: string }> = ({ boardId, snapshotName }) => {
  const navigate = useNavigate();
  const { data: views = [] } = useBoardViews(boardId);

  // ActiveViewId persiste por boardId em localStorage
  const storageKey = `lfpro-database-active-view-${boardId}`;
  const [activeViewId, setActiveViewIdState] = useState<string | null>(() => {
    try { return localStorage.getItem(storageKey); } catch { return null; }
  });
  const setActiveViewId = (id: string) => {
    setActiveViewIdState(id);
    try { localStorage.setItem(storageKey, id); } catch {}
  };

  // Quando views carregam, se activeViewId nao existe em views, fallback pra default
  useEffect(() => {
    if (views.length === 0) return;
    const exists = activeViewId && views.some((v) => v.id === activeViewId);
    if (!exists) {
      const def = views.find((v) => v.is_default) ?? views[0];
      setActiveViewIdState(def.id);
    }
  }, [views, activeViewId]);

  if (!boardId) {
    return (
      <div contentEditable={false} className="my-3 border border-destructive/30 rounded-md p-4 bg-destructive/5 text-sm text-destructive">
        Database invalida (boardId ausente).
      </div>
    );
  }

  return (
    <div contentEditable={false} className="my-3 border border-border rounded-md bg-card overflow-hidden">
      <div className="flex items-center gap-2 px-3 py-2 border-b border-border bg-muted/30">
        <Database className="w-4 h-4 text-primary" />
        <button
          onClick={() => navigate(`/board/${boardId}`)}
          className="font-semibold text-sm text-foreground hover:underline"
          title="Abrir database em tela cheia"
        >
          {snapshotName || 'Database'}
        </button>
      </div>
      <DatabaseViewTabs
        boardId={boardId}
        activeViewId={activeViewId}
        onChangeView={setActiveViewId}
      />
      <DatabaseViewRenderer boardId={boardId} activeViewId={activeViewId} />
    </div>
  );
};
```
