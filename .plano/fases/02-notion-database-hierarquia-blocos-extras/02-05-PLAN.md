# Fase 02 Plano 02-05: Bloco Database no BlockNote (criacao e render inline)

**Objetivo:** Adicionar bloco `database` ao schema BlockNote e ao slash menu pt-BR. Ao escolher `/database`, abrir dialog pra nome+icone, criar boards/groups/columns/views, e inserir bloco com `props.boardId`. Render do bloco e stub (header + placeholder). Views completas vem em 02-06+.

## Pesquisa inline aplicada

- **BlockNote custom blocks:** `createReactBlockSpec({ type, content: 'none', propSchema: { boardId: { type: 'string', default: '' }, snapshotName: { type: 'string', default: '' } } }, { render: (props) => ... })`. API v0.51 retorna factory `(options?) => BlockSpec` - invocar como `DatabaseBlock()` em blocnote-schema (mesmo padrao do EmbedBoardBlock).
- **content:'none':** Bloco atomico, sem texto editavel dentro. `contentEditable={false}` no wrapper externo.
- **Editor.updateBlock:** Pode atualizar props depois (ex: trocar boardId). MVP nao precisa.

```typescript
import { firstKey } from '@/utils/lexorank';

/**
 * Cria database inline ancorada numa page.
 *
 * Operacao em sequencia:
 *  1. INSERT em boards com page_id = pageId
 *  2. INSERT em groups (1 grupo default "Itens")
 *  3. INSERT em columns (Status, Date, People - 3 colunas iniciais)
 *  4. INSERT em board_views (4 views: Tabela, Kanban, Calendario, Lista detalhada)
 *
 * Retorna { boardId, viewIds } pra que o componente inserindo o bloco use boardId
 * em props.boardId.
 *
 * Status default: 3 labels (Em andamento, Concluido, Pendente) em hsl warm.
 */
export const useCreateDatabase = () => {
  const qc = useQueryClient();
  return useMutation({
    retry: 1,
    mutationFn: async ({
      workspaceId,
      pageId,
      name,
      icon,
    }: {
      workspaceId: string;
      pageId: string;
      name: string;
      icon?: string;
    }) => {
      const { data: user } = await supabase.auth.getUser();
      if (!user.user) throw new Error('Not authenticated');

      // 1. Board (database inline)
      const { data: board, error: boardError } = await supabase
        .from('boards')
        .insert({
          workspace_id: workspaceId,
          name,
          icon: icon ?? null,
          state: 'active',
          page_id: pageId,
          created_by: user.user.id,
        } as never)
        .select()
        .single();
      if (boardError) throw boardError;
      const boardId = (board as { id: string }).id;

      // 2. Grupo default
      const { error: groupError } = await supabase.from('groups').insert({
        board_id: boardId,
        title: 'Itens',
        color: '#a89172', // warm gold
        position: 0,
        is_collapsed: false,
      } as never);
      if (groupError) throw groupError;

      // 3. Colunas iniciais (Status, Date, People)
      const initialColumns = [
        {
          board_id: boardId,
          column_type: 'status',
          title: 'Status',
          position: 0,
          width: 140,
          settings: {
            labels: {
              key1: { name: 'Pendente', color: '#9CA3AF' },
              key2: { name: 'Em andamento', color: '#F59E0B' },
              key3: { name: 'Concluido', color: '#10B981', isDone: true },
            },
          },
        },
        {
          board_id: boardId,
          column_type: 'date',
          title: 'Data',
          position: 1,
          width: 120,
          settings: {},
        },
        {
          board_id: boardId,
          column_type: 'people',
          title: 'Responsavel',
          position: 2,
          width: 140,
          settings: {},
        },
      ];
      const { error: colError } = await supabase.from('columns').insert(initialColumns as never);
      if (colError) throw colError;

      // 4. 4 views (Tabela, Kanban, Calendario, Lista detalhada)
      const initialViews = [
        { board_id: boardId, name: 'Tabela', view_type: 'table', position: 0, is_default: true, config: {} },
        { board_id: boardId, name: 'Kanban', view_type: 'kanban', position: 1, is_default: false, config: {} },
        { board_id: boardId, name: 'Calendario', view_type: 'calendar', position: 2, is_default: false, config: {} },
        { board_id: boardId, name: 'Lista detalhada', view_type: 'list_detailed', position: 3, is_default: false, config: { visibleProps: ['status', 'date', 'people'] } },
      ];
      const { data: views, error: viewError } = await supabase
        .from('board_views')
        .insert(initialViews.map((v) => ({ ...v, created_by: user.user.id })) as never)
        .select();
      if (viewError) throw viewError;

      return {
        boardId,
        viewIds: (views ?? []).map((v: { id: string }) => v.id),
        defaultViewId: (views ?? []).find((v: { is_default: boolean }) => v.is_default)?.id,
      };
    },
    onSuccess: (_data, vars) => {
      qc.invalidateQueries({ queryKey: ['boards'] });
      qc.invalidateQueries({ queryKey: ['all-boards'] });
      qc.invalidateQueries({ queryKey: ['databases-for-page', vars.pageId] });
      qc.invalidateQueries({ queryKey: ['pages-tree'] });
    },
  });
};
```

```typescript
import React, { useState } from 'react';
import { Dialog, DialogContent, DialogHeader, DialogTitle, DialogFooter } from '@/components/ui/dialog';
import { Button } from '@/components/ui/button';
import { Input } from '@/components/ui/input';
import { EmojiColorPicker } from '@/components/shared/EmojiColorPicker';
import { useCreateDatabase } from '@/hooks/useCrudMutations';
import { toast } from 'sonner';

interface CreateDatabaseDialogProps {
  open: boolean;
  onOpenChange: (open: boolean) => void;
  workspaceId: string;
  pageId: string;
  onCreated: (result: { boardId: string; name: string }) => void;
}

const CreateDatabaseDialog: React.FC<CreateDatabaseDialogProps> = ({
  open, onOpenChange, workspaceId, pageId, onCreated,
}) => {
  const [name, setName] = useState('');
  const [icon, setIcon] = useState<string | undefined>(undefined);
  const [isCreating, setIsCreating] = useState(false);
  const createDatabase = useCreateDatabase();

  const handleConfirm = async () => {
    const trimmed = name.trim();
    if (!trimmed) {
      toast.error('Informe um nome pra database');
      return;
    }
    setIsCreating(true);
    try {
      const result = await createDatabase.mutateAsync({
        workspaceId, pageId, name: trimmed, icon,
      });
      onCreated({ boardId: result.boardId, name: trimmed });
      toast.success('Database criada');
      // reset
      setName('');
      setIcon(undefined);
      onOpenChange(false);
    } catch (e) {
      toast.error('Erro ao criar database. Tente novamente.');
    } finally {
      setIsCreating(false);
    }
  };

  const handleOpenChange = (next: boolean) => {
    if (!next) { setName(''); setIcon(undefined); }
    onOpenChange(next);
  };

  return (
    <Dialog open={open} onOpenChange={handleOpenChange}>
      <DialogContent className="max-w-md">
        <DialogHeader>
          <DialogTitle>Nova database</DialogTitle>
        </DialogHeader>
        <div className="space-y-3">
          <div className="flex items-center gap-2">
            <EmojiColorPicker
              currentEmoji={icon}
              currentColor={null}
              onSelectEmoji={setIcon}
              onSelectColor={() => {}}
            />
            <Input
              value={name}
              onChange={(e) => setName(e.target.value)}
              placeholder="Nome da database"
              autoFocus
              onKeyDown={(e) => { if (e.key === 'Enter') handleConfirm(); }}
            />
          </div>
          <p className="text-xs text-muted-foreground">
            A database e criada com 3 colunas (Status, Data, Responsavel), um grupo padrao e 4 views (Tabela, Kanban, Calendario, Lista detalhada).
          </p>
        </div>
        <DialogFooter>
          <Button variant="outline" onClick={() => handleOpenChange(false)} disabled={isCreating}>
            Cancelar
          </Button>
          <Button onClick={handleConfirm} disabled={isCreating}>
            {isCreating ? 'Criando...' : 'Criar database'}
          </Button>
        </DialogFooter>
      </DialogContent>
    </Dialog>
  );
};

export default CreateDatabaseDialog;
```

```typescript
import React from 'react';
import { createReactBlockSpec } from '@blocknote/react';
import { Database } from 'lucide-react';
import { useNavigate } from 'react-router-dom';
import DatabaseViewTabs from '@/components/database/DatabaseViewTabs';
import DatabaseViewRenderer from '@/components/database/DatabaseViewRenderer';

/**
 * Bloco database inline. Renderiza:
 *  - Header: icone Database + nome do board + (futuro) menu
 *  - DatabaseViewTabs (02-08): tabs de view + criar view
 *  - DatabaseViewRenderer (02-06/07): renderiza view ativa (Tabela/Kanban/Calendar/ListDetailed)
 *
 * Props:
 *  - boardId: id do board (boards.page_id IS NOT NULL) ancorado nesta page
 *  - snapshotName: nome no momento do insert (fallback se board for deletado)
 *
 * content: 'none' (bloco atomico, sem texto editavel dentro)
 *
 * IMPORTANTE: createReactBlockSpec retorna factory na API v0.51; invocar como
 * DatabaseBlock() em blocnote-schema.ts.
 */
export const DatabaseBlock = createReactBlockSpec(
  {
    type: 'database',
    content: 'none',
    propSchema: {
      boardId: { type: 'string', default: '' },
      snapshotName: { type: 'string', default: '' },
    },
  },
  {
    render: (props) => {
      const boardId = props.block.props.boardId as string;
      const snapshotName = props.block.props.snapshotName as string;
      return (
        <DatabaseBlockView boardId={boardId} snapshotName={snapshotName} />
      );
    },
  },
);

const DatabaseBlockView: React.FC<{ boardId: string; snapshotName: string }> = ({ boardId, snapshotName }) => {
  const navigate = useNavigate();

  if (!boardId) {
    return (
      <div contentEditable={false} className="my-3 border border-destructive/30 rounded-md p-4 bg-destructive/5 text-sm text-destructive">
        Database invalida (boardId ausente).
      </div>
    );
  }

  return (
    <div
      contentEditable={false}
      className="my-3 border border-border rounded-md bg-card overflow-hidden"
    >
      <div className="flex items-center gap-2 px-3 py-2 border-b border-border bg-muted/30">
        <Database className="w-4 h-4 text-primary" />
        <button
          onClick={() => navigate(`/board/${boardId}`)}
          className="font-semibold text-sm text-foreground hover:underline"
          title="Abrir database em tela cheia"
        >
          {snapshotName || 'Database'}
        </button>
      </div>
      <DatabaseViewTabs boardId={boardId} />
      <DatabaseViewRenderer boardId={boardId} />
    </div>
  );
};
```
