# Fase 02 Plano 02-07: DatabaseListView (Notion list view)

**Objetivo:** Implementar o unico componente de view novo da Fase 02. Cada item renderiza como linha grande com titulo destacado + chips empilhados das props visiveis. Config de quais props mostrar persiste em board_views.config.

## Pesquisa inline aplicada

- **Notion list view:** Layout vertical. Cada item: titulo grande + linha horizontal de "metadados" (status pill, due date, assignee avatars, tags). Sem grid de colunas. Click no titulo abre painel de detalhes.
- **Config persistente:** `board_views.config jsonb` ja existe. Em vez de criar nova coluna, usar `config.visibleProps: string[]` (array de column_ids).
- **Padrao de cell render:** Reusar componentes `*Cell.tsx` em modo readonly (passar `onChange` que nao faz nada) ou criar mini renderers proprios pra cada tipo (mais leve). Optamos por mini renderers, similar ao CellRender do EmbedBoardBlock.

@src/components/board/BoardTable.tsx - usa useApp() para items, columns, groups, columnValues

@src/components/page/blocks/EmbedBoardBlock.tsx - CellRender minimalista por tipo (modelo de mini renderers)

```typescript
export const useUpdateBoardViewConfig = () => {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: async ({ viewId, config }: { viewId: string; config: Record<string, unknown> }) => {
      const { error } = await supabase
        .from('board_views')
        .update({ config })
        .eq('id', viewId);
      if (error) throw error;
    },
    onSuccess: () => {
      qc.invalidateQueries({ queryKey: ['board_views'] });
    },
  });
};
```

```typescript
import React from 'react';
import { format, parseISO } from 'date-fns';
import { ptBR } from 'date-fns/locale';
import { Calendar, User } from 'lucide-react';
import type { Column, Item, ColumnValue } from '@/types/board';

interface DatabaseListItemProps {
  item: Item;
  visibleColumns: Column[]; // colunas selecionadas em config.visibleProps
  columnValues: Record<string, ColumnValue>; // map column_id -> { value }
  profiles: Array<{ id: string; name: string; avatar_url?: string | null }>;
  onClick: (item: Item) => void;
}

/**
 * Renderiza UM item da DatabaseListView (Notion-style).
 *
 * Layout:
 *  - Linha 1: titulo grande (text-lg font-heading-medium), clicavel
 *  - Linha 2: chips horizontais das visibleColumns (status pill, data, avatares people, etc.)
 *  - py-3 border-b border-border
 */
const DatabaseListItem: React.FC<DatabaseListItemProps> = ({
  item, visibleColumns, columnValues, profiles, onClick,
}) => {
  return (
    <div className="py-3 border-b border-border last:border-b-0 hover:bg-muted/30 transition-colors px-3 cursor-pointer" onClick={() => onClick(item)}>
      <h3 className="font-heading text-lg font-medium text-foreground mb-1.5 truncate" title={item.name}>
        {item.name}
      </h3>
      <div className="flex items-center gap-2 flex-wrap text-xs">
        {visibleColumns.map((col) => {
          const cv = columnValues[col.id];
          if (!cv || cv.value === null || cv.value === undefined) return null;
          return (
            <ListItemPropChip
              key={col.id}
              column={col}
              value={cv.value}
              profiles={profiles}
            />
          );
        })}
      </div>
    </div>
  );
};

/**
 * Render minimalista de uma propriedade por tipo de coluna (subset 8 tipos do MVP).
 * Esses chips sao read-only (click no item abre ItemDetailPanel pra editar).
 */
const ListItemPropChip: React.FC<{
  column: Column;
  value: unknown;
  profiles: Array<{ id: string; name: string; avatar_url?: string | null }>;
}> = ({ column, value, profiles }) => {
  const settings = column.settings ?? {};

  switch (column.type) {
    case 'status': {
      const key = String(value);
      const label = (settings.labels ?? {})[key];
      if (!label) return null;
      return (
        <span
          className="inline-flex items-center gap-1.5 px-2 py-0.5 rounded-full text-xs font-medium"
          style={{ backgroundColor: `${label.color}22`, color: label.color }}
        >
          <span className="w-1.5 h-1.5 rounded-full" style={{ backgroundColor: label.color }} />
          {label.name}
        </span>
      );
    }
    case 'date': {
      const dateStr = typeof value === 'string' ? value : (value as { date?: string })?.date;
      if (!dateStr) return null;
      try {
        const d = parseISO(dateStr);
        return (
          <span className="inline-flex items-center gap-1 text-muted-foreground">
            <Calendar className="w-3 h-3" />
            {format(d, 'dd MMM yyyy', { locale: ptBR })}
          </span>
        );
      } catch { return null; }
    }
    case 'people': {
      const ids = Array.isArray(value) ? (value as string[]) : [];
      const people = ids.map((id) => profiles.find((p) => p.id === id)).filter(Boolean);
      if (people.length === 0) return null;
      return (
        <span className="inline-flex items-center gap-1">
          <span className="flex -space-x-1.5">
            {people.slice(0, 3).map((p, i) => (
              <span
                key={p!.id}
                className="w-5 h-5 rounded-full border-2 border-background bg-primary/20 text-primary text-[10px] flex items-center justify-center font-semibold"
                title={p!.name}
              >
                {p!.name.charAt(0).toUpperCase()}
              </span>
            ))}
          </span>
          {people.length > 3 && (
            <span className="text-muted-foreground">+{people.length - 3}</span>
          )}
        </span>
      );
    }
    case 'number': {
      const n = typeof value === 'number' ? value : Number(value);
      if (Number.isNaN(n)) return null;
      return (
        <span className="font-tabular-nums text-muted-foreground">
          {n.toLocaleString('pt-BR')}{settings.unit ? ` ${settings.unit}` : ''}
        </span>
      );
    }
    case 'checkbox': {
      const checked = Boolean(value);
      return (
        <span className={`inline-flex items-center gap-1 px-1.5 py-0.5 rounded ${checked ? 'bg-primary/15 text-primary' : 'bg-muted text-muted-foreground'}`}>
          {checked ? 'Sim' : 'Nao'}
        </span>
      );
    }
    case 'dropdown': {
      const v = String(value);
      if (!v) return null;
      return <span className="px-2 py-0.5 rounded bg-muted text-foreground">{v}</span>;
    }
    case 'text':
    case 'long_text': {
      const text = String(value);
      if (!text) return null;
      return <span className="text-muted-foreground truncate max-w-[200px]" title={text}>{text}</span>;
    }
    default:
      return null;
  }
};

export default DatabaseListItem;
```

```typescript
import React from 'react';
import { Popover, PopoverContent, PopoverTrigger } from '@/components/ui/popover';
import { Settings2 } from 'lucide-react';
import { Checkbox } from '@/components/ui/checkbox';
import { useUpdateBoardViewConfig } from '@/hooks/useBoardViews';
import type { Column } from '@/types/board';
import { toast } from 'sonner';

interface Props {
  viewId: string;
  columns: Column[];
  visibleProps: string[]; // ids das columns visiveis
}

const DatabaseListViewConfig: React.FC<Props> = ({ viewId, columns, visibleProps }) => {
  const updateConfig = useUpdateBoardViewConfig();

  const toggleProp = async (columnId: string) => {
    const newVisible = visibleProps.includes(columnId)
      ? visibleProps.filter((id) => id !== columnId)
      : [...visibleProps, columnId];
    try {
      await updateConfig.mutateAsync({ viewId, config: { visibleProps: newVisible } });
    } catch {
      toast.error('Erro ao salvar config da view');
    }
  };

  return (
    <Popover>
      <PopoverTrigger asChild>
        <button
          className="p-1.5 rounded hover:bg-muted text-muted-foreground transition-colors"
          aria-label="Configurar props visiveis"
          title="Configurar props visiveis"
        >
          <Settings2 className="w-3.5 h-3.5" />
        </button>
      </PopoverTrigger>
      <PopoverContent className="w-56 p-2" align="end">
        <p className="text-xs font-semibold text-muted-foreground uppercase tracking-wider px-2 pb-1.5">
          Props visiveis
        </p>
        <div className="space-y-0.5">
          {columns.map((col) => (
            <label
              key={col.id}
              className="flex items-center gap-2 px-2 py-1.5 rounded hover:bg-muted cursor-pointer text-sm"
            >
              <Checkbox
                checked={visibleProps.includes(col.id)}
                onCheckedChange={() => toggleProp(col.id)}
              />
              <span className="truncate flex-1">{col.title}</span>
            </label>
          ))}
        </div>
      </PopoverContent>
    </Popover>
  );
};

export default DatabaseListViewConfig;
```

```typescript
import React, { useMemo } from 'react';
import { useApp } from '@/context/AppContext';
import { useBoardViews } from '@/hooks/useBoardViews';
import { useProfiles } from '@/hooks/useSupabaseData';
import DatabaseListItem from './DatabaseListItem';
import DatabaseListViewConfig from './DatabaseListViewConfig';
import type { Item, Group, Column, ColumnValue } from '@/types/board';

interface Props {
  mode?: 'board' | 'database';
}

/**
 * View Notion-style "lista detalhada".
 *
 * Le do BoardProvider local (via useApp): activeBoardId, items, columns, groups, columnValues.
 * Agrupa items por group (igual BoardTable). Cada item renderiza com titulo grande
 * + chips das props visiveis.
 *
 * Config persiste em board_views.config.visibleProps (array de column_ids).
 */
const DatabaseListView: React.FC<Props> = ({ mode = 'database' }) => {
  const { activeBoardId, items, columns, groups, columnValues, setSelectedItem } = useApp();
  const { data: views = [] } = useBoardViews(activeBoardId);
  const { data: profiles = [] } = useProfiles();

  // View ativa = a unica view_type='list_detailed' deste board (ou primeira)
  const listView = views.find((v) => v.view_type === 'list_detailed');
  const config = (listView?.config ?? {}) as { visibleProps?: string[] };

  // Default: status + date + people (column_ids resolvidos via title fallback)
  const visibleProps = useMemo(() => {
    if (Array.isArray(config.visibleProps) && config.visibleProps.length > 0) {
      return config.visibleProps;
    }
    // Default: pegar primeira coluna de cada tipo desejado
    const defaults = ['status', 'date', 'people']
      .map((t) => columns.find((c) => c.type === t)?.id)
      .filter(Boolean) as string[];
    return defaults;
  }, [config.visibleProps, columns]);

  const visibleColumns = useMemo(
    () => visibleProps.map((id) => columns.find((c) => c.id === id)).filter(Boolean) as Column[],
    [visibleProps, columns],
  );

  // Map column_values: item_id -> column_id -> { value }
  const cvByItem = useMemo(() => {
    const map: Record<string, Record<string, ColumnValue>> = {};
    for (const cv of columnValues) {
      if (!map[cv.item_id]) map[cv.item_id] = {};
      map[cv.item_id][cv.column_id] = cv;
    }
    return map;
  }, [columnValues]);

  // Items agrupados
  const itemsByGroup = useMemo(() => {
    const map: Record<string, Item[]> = {};
    for (const item of items) {
      const gid = item.groupId ?? 'no-group';
      if (!map[gid]) map[gid] = [];
      map[gid].push(item);
    }
    // ordenar por position
    for (const gid of Object.keys(map)) {
      map[gid].sort((a, b) => a.position - b.position);
    }
    return map;
  }, [items]);

  return (
    <div className={mode === 'database' ? 'max-h-[560px] overflow-y-auto' : 'h-full overflow-y-auto'}>
      <div className="flex items-center justify-between px-3 py-2 border-b border-border bg-muted/20">
        <span className="text-xs font-semibold text-muted-foreground uppercase tracking-wider">
          {items.length} {items.length === 1 ? 'item' : 'itens'}
        </span>
        {listView && (
          <DatabaseListViewConfig
            viewId={listView.id}
            columns={columns}
            visibleProps={visibleProps}
          />
        )}
      </div>
      {groups.length === 0 && items.length === 0 && (
        <p className="px-3 py-8 text-center text-sm text-muted-foreground">
          Nenhum item ainda. Use a view Tabela pra adicionar.
        </p>
      )}
      {groups.map((g) => {
        const groupItems = itemsByGroup[g.id] ?? [];
        if (groupItems.length === 0) return null;
        return (
          <div key={g.id}>
            <div className="px-3 py-1.5 bg-muted/40 text-xs font-semibold uppercase tracking-wider flex items-center gap-2" style={{ color: g.color ?? undefined }}>
              <span className="w-1.5 h-1.5 rounded-full" style={{ backgroundColor: g.color ?? '#a89172' }} />
              {g.title}
            </div>
            {groupItems.map((item) => (
              <DatabaseListItem
                key={item.id}
                item={item}
                visibleColumns={visibleColumns}
                columnValues={cvByItem[item.id] ?? {}}
                profiles={profiles}
                onClick={(i) => setSelectedItem(i)}
              />
            ))}
          </div>
        );
      })}
    </div>
  );
};

export default DatabaseListView;
```
