# Fase 02 Plano 02-10: Bloco Synced (synced_blocks)

**Objetivo:** Permitir conteudo compartilhado entre multiplas pages do mesmo workspace via synced blocks. Editar em qualquer instancia reflete em todas. Mini-editor BlockNote interno por bloco.

## Pesquisa inline aplicada

- **BlockNote nested editors:** Cada synced block renderiza um BlockNote editor isolado (com schema mais simples ou o mesmo lfproBlockNoteSchema). API: `useCreateBlockNote({ initialContent })` dentro do `render`.
- **Realtime sync:** Subscription em `synced_blocks` invalida `['synced-block', id]`. React Query refetch. BlockNote `editor.replaceBlocks` aplica o novo content (se nao for o bloco que originou a edicao, pra evitar overwrite local pendente).
- **Last-write-wins:** Sem CRDT no MVP. Cada save sobrescreve o content. Edicao concorrente pode sobrescrever a outra (aceito no MVP).

```typescript
import type { SyncedBlock } from '@/types/page';

/**
 * Query individual de synced_block.
 * staleTime: 0 (sempre fresh apos invalidacao via realtime).
 * Se RLS bloquear (member de outro workspace), retorna null (queryFn captura).
 */
export const useSyncedBlock = (id: string | null | undefined) =>
  useQuery({
    queryKey: ['synced-block', id],
    enabled: !!id,
    staleTime: 0,
    queryFn: async () => {
      const { data, error } = await supabase
        .from('synced_blocks')
        .select('id, workspace_id, content, created_by, created_at, updated_at')
        .eq('id', id!)
        .maybeSingle();
      if (error) {
        // RLS pode bloquear leitura -> retorna null pra UI mostrar placeholder
        return null;
      }
      return (data as SyncedBlock | null) ?? null;
    },
  });

/**
 * Lista synced_blocks de um workspace pro picker.
 */
export const useSyncedBlocksByWorkspace = (workspaceId: string | null | undefined, enabled: boolean = true) =>
  useQuery({
    queryKey: ['synced-blocks-workspace', workspaceId],
    enabled: !!workspaceId && enabled,
    staleTime: 60_000,
    queryFn: async () => {
      const { data, error } = await supabase
        .from('synced_blocks')
        .select('id, workspace_id, content, created_at, updated_at')
        .eq('workspace_id', workspaceId!)
        .order('updated_at', { ascending: false });
      if (error) throw error;
      return data ?? [];
    },
  });
```

```typescript
export const useCreateSyncedBlock = () => {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: async ({ workspaceId, content }: { workspaceId: string; content: unknown[] }) => {
      const { data: user } = await supabase.auth.getUser();
      if (!user.user) throw new Error('Not authenticated');
      const { data, error } = await supabase
        .from('synced_blocks')
        .insert({
          workspace_id: workspaceId,
          content,
          created_by: user.user.id,
        } as never)
        .select()
        .single();
      if (error) throw error;
      return data;
    },
    onSuccess: (_d, vars) => {
      qc.invalidateQueries({ queryKey: ['synced-blocks-workspace', vars.workspaceId] });
    },
  });
};

export const useUpdateSyncedBlockContent = () => {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: async ({ id, content }: { id: string; content: unknown[] }) => {
      const { error } = await supabase
        .from('synced_blocks')
        .update({ content } as never)
        .eq('id', id);
      if (error) throw error;
    },
    onSuccess: (_d, vars) => {
      qc.invalidateQueries({ queryKey: ['synced-block', vars.id] });
    },
  });
};
```

```typescript
import React, { useEffect, useRef, useMemo } from 'react';
import { createReactBlockSpec } from '@blocknote/react';
import { useCreateBlockNote } from '@blocknote/react';
import { BlockNoteView } from '@blocknote/mantine';
import { pt as ptDictionary } from '@blocknote/core/locales';
import { Repeat2 } from 'lucide-react';
import { lfproBlockNoteLightTheme, lfproBlockNoteDarkTheme } from '../blocknote-theme';
import { useTheme } from 'next-themes';
import { useSyncedBlock } from '@/hooks/useSupabaseData';
import { useUpdateSyncedBlockContent } from '@/hooks/useCrudMutations';
import type { PartialBlock } from '@blocknote/core';

const SAVE_DEBOUNCE_MS = 1000;

export const SyncedBlock = createReactBlockSpec(
  {
    type: 'synced',
    content: 'none',
    propSchema: {
      synced_block_id: { type: 'string', default: '' },
    },
  },
  {
    render: (props) => {
      const id = props.block.props.synced_block_id as string;
      return <SyncedBlockView id={id} />;
    },
  },
);

const SyncedBlockView: React.FC<{ id: string }> = ({ id }) => {
  const { resolvedTheme } = useTheme();
  const { data: syncedBlock, isLoading } = useSyncedBlock(id);
  const updateContent = useUpdateSyncedBlockContent();
  const saveTimerRef = useRef<ReturnType<typeof setTimeout> | null>(null);
  const lastSavedRef = useRef<string>('');
  const isLocalChangeRef = useRef(false);

  // Editor BlockNote interno (schema basico, sem mention/embed/database aninhado pra MVP)
  const initialContent = useMemo(() => {
    if (!syncedBlock?.content) return undefined;
    return Array.isArray(syncedBlock.content)
      ? (syncedBlock.content as unknown as PartialBlock[])
      : undefined;
  }, [syncedBlock?.content]);

  const editor = useCreateBlockNote({
    dictionary: ptDictionary,
    initialContent,
  });

  const theme = resolvedTheme === 'dark' ? lfproBlockNoteDarkTheme : lfproBlockNoteLightTheme;

  // Quando content externo muda (via realtime), aplicar no editor SE nao for a edicao local que disparou
  useEffect(() => {
    if (!syncedBlock?.content) return;
    const serverContent = JSON.stringify(syncedBlock.content);
    if (serverContent === lastSavedRef.current) return; // sem mudanca real
    if (isLocalChangeRef.current) {
      // ignora; foi nossa propria edicao confirmada
      isLocalChangeRef.current = false;
      lastSavedRef.current = serverContent;
      return;
    }
    // Update externo: aplicar ao editor
    try {
      editor.replaceBlocks(editor.document, syncedBlock.content as PartialBlock[]);
      lastSavedRef.current = serverContent;
    } catch (e) {
      console.error('replaceBlocks erro:', e);
    }
  }, [syncedBlock?.content, editor]);

  const handleChange = () => {
    if (!id) return;
    if (saveTimerRef.current) clearTimeout(saveTimerRef.current);
    saveTimerRef.current = setTimeout(async () => {
      const content = editor.document;
      const serialized = JSON.stringify(content);
      if (serialized === lastSavedRef.current) return;
      isLocalChangeRef.current = true;
      lastSavedRef.current = serialized;
      try {
        await updateContent.mutateAsync({ id, content });
      } catch (e) {
        console.error('save synced block erro:', e);
      }
    }, SAVE_DEBOUNCE_MS);
  };

  if (!id) {
    return (
      <div contentEditable={false} className="my-3 border border-destructive/30 rounded-md p-3 text-sm text-destructive bg-destructive/5">
        Bloco sincronizado invalido (id ausente).
      </div>
    );
  }

  if (!isLoading && !syncedBlock) {
    return (
      <div contentEditable={false} className="my-3 border border-border rounded-md p-3 text-sm text-muted-foreground bg-muted/30">
        Bloco nao acessivel (workspace diferente ou removido).
      </div>
    );
  }

  return (
    <div contentEditable={false} className="my-3 border-l-2 border-primary/40 rounded-md bg-muted/10 overflow-hidden">
      <div className="flex items-center gap-1.5 px-3 py-1 text-xs text-muted-foreground border-b border-border/50">
        <Repeat2 className="w-3 h-3" />
        <span>Bloco sincronizado</span>
      </div>
      <div className="px-2 py-1">
        <BlockNoteView
          editor={editor}
          theme={theme}
          onChange={handleChange}
        />
      </div>
    </div>
  );
};
```

```typescript
import React, { useState } from 'react';
import { Dialog, DialogContent, DialogHeader, DialogTitle, DialogFooter } from '@/components/ui/dialog';
import { Button } from '@/components/ui/button';
import { useSyncedBlocksByWorkspace } from '@/hooks/useSupabaseData';
import { useCreateSyncedBlock } from '@/hooks/useCrudMutations';
import { Repeat2, Plus } from 'lucide-react';
import { toast } from 'sonner';
import { format, parseISO } from 'date-fns';
import { ptBR } from 'date-fns/locale';

interface Props {
  open: boolean;
  onOpenChange: (open: boolean) => void;
  workspaceId: string;
  onSelect: (syncedBlockId: string) => void;
}

const SyncedBlockPickerDialog: React.FC<Props> = ({ open, onOpenChange, workspaceId, onSelect }) => {
  const { data: synced = [] } = useSyncedBlocksByWorkspace(workspaceId, open);
  const createSynced = useCreateSyncedBlock();
  const [isCreating, setIsCreating] = useState(false);

  const handleCreateNew = async () => {
    setIsCreating(true);
    try {
      const created = await createSynced.mutateAsync({
        workspaceId,
        content: [{ type: 'paragraph', content: 'Conteudo sincronizado' }],
      });
      onSelect((created as { id: string }).id);
      onOpenChange(false);
      toast.success('Bloco sincronizado criado');
    } catch {
      toast.error('Erro ao criar bloco sincronizado');
    } finally {
      setIsCreating(false);
    }
  };

  const previewText = (content: unknown): string => {
    if (!Array.isArray(content) || content.length === 0) return '(vazio)';
    const first = content[0] as { content?: Array<{ text?: string }> };
    if (Array.isArray(first.content)) {
      return first.content.map((c) => c.text ?? '').join('').slice(0, 80);
    }
    return '(sem texto)';
  };

  return (
    <Dialog open={open} onOpenChange={onOpenChange}>
      <DialogContent className="max-w-lg">
        <DialogHeader>
          <DialogTitle>Bloco sincronizado</DialogTitle>
        </DialogHeader>
        <Button onClick={handleCreateNew} disabled={isCreating} className="w-full" variant="default">
          <Plus className="w-4 h-4 mr-2" />
          {isCreating ? 'Criando...' : 'Criar novo bloco sincronizado'}
        </Button>
        {synced.length > 0 && (
          <>
            <p className="text-xs font-semibold text-muted-foreground uppercase tracking-wider pt-2">
              Ou referencie um existente
            </p>
            <div className="max-h-60 overflow-y-auto space-y-1">
              {synced.map((sb) => (
                <button
                  key={sb.id}
                  onClick={() => { onSelect(sb.id); onOpenChange(false); }}
                  className="w-full flex items-start gap-2 p-2 rounded hover:bg-muted transition-colors text-left"
                >
                  <Repeat2 className="w-3.5 h-3.5 mt-0.5 text-muted-foreground" />
                  <div className="flex-1 min-w-0">
                    <p className="text-sm text-foreground truncate">{previewText(sb.content)}</p>
                    <p className="text-xs text-muted-foreground">
                      Atualizado em {format(parseISO(sb.updated_at), 'dd MMM yyyy', { locale: ptBR })}
                    </p>
                  </div>
                </button>
              ))}
            </div>
          </>
        )}
        <DialogFooter>
          <Button variant="outline" onClick={() => onOpenChange(false)}>Fechar</Button>
        </DialogFooter>
      </DialogContent>
    </Dialog>
  );
};

export default SyncedBlockPickerDialog;
```
