# Fase 01 Plano 01-08: Presence + realtime sync + image upload

**Objetivo:** Completar o pacote colaborativo MVP. Imagens funcionando no editor (drag/drop, paste, slash menu), refresh automatico do conteudo quando outro usuario salva, e indicador visual de quem esta editando agora.

## Achados de Pesquisa (Inline)

- **BlockNote uploadFile API:** O hook `useCreateBlockNote` aceita opcao `uploadFile: (file: File) => Promise<string>`. A string retornada e a URL final usada no bloco `image`. Sem hooks externos.
- **Supabase Storage upload pattern (do projeto):** ver `src/hooks/useFileUpload.ts` para padrao do bucket `attachments`. Path convention: `{userId}/{itemId}/{fileName}`. Vamos usar `{userId}/page-{pageId}/{timestamp}-{filename}` para distinguir.
- **Presence channel pattern:** ver `src/hooks/usePresence.ts` (linha 11) — usa `supabase.channel('sidebar-presence', { config: { presence: { key: 'sidebar' } } })`. Para page-specific vamos usar `page-presence-${pageId}`.
- **Realtime de pages:** o projeto ja faz `useRealtimeSync` em workspace-sync mas so para tabelas de boards. Adicionar `pages` ao listener com debounce. CUIDADO: invalidar `['page', pageId]` so re-fetch e o auto-save com `replaceBlocks` poderia descartar edicoes locais. Solucao: cache somente atualiza se autoSave.status === 'idle' OU 'saved' (sem mudancas pendentes). Implementacao simples: no Page.tsx, monitorar `updated_at` da query e so chamar replaceBlocks se for de outro usuario E nao houver edicoes locais pendentes (autoSave.status !== 'pending' && !== 'saving').

```typescript
import { useCallback } from 'react';
import { supabase } from '@/integrations/supabase/client';

/**
 * Upload de imagem para o bucket `attachments` do Supabase Storage.
 * Path: {userId}/page-{pageId}/{timestamp}-{safeFilename}
 * Retorna URL publica para uso no bloco image do BlockNote.
 *
 * Reusa o bucket existente (ver supabase/migrations/20260408211000_create_attachments_bucket.sql).
 * Sem novo bucket por decisao do CONTEXT.md.
 */
export function usePageImageUpload(pageId: string) {
  const uploadImage = useCallback(
    async (file: File): Promise<string> => {
      const { data: authData } = await supabase.auth.getUser();
      const userId = authData.user?.id;
      if (!userId) throw new Error('Usuario nao autenticado');

      // Sanitize filename: remove acentos, espacos, caracteres especiais
      const safeName = file.name
        .normalize('NFD')
        .replace(/[̀-ͯ]/g, '')
        .replace(/[^a-zA-Z0-9._-]/g, '-')
        .toLowerCase();
      const timestamp = Date.now();
      const path = `${userId}/page-${pageId}/${timestamp}-${safeName}`;

      const { error } = await supabase.storage
        .from('attachments')
        .upload(path, file, {
          cacheControl: '3600',
          upsert: false,
        });
      if (error) throw error;

      const { data } = supabase.storage.from('attachments').getPublicUrl(path);
      return data.publicUrl;
    },
    [pageId]
  );

  return uploadImage;
}
```

Hook de presence por page.

```typescript
import { useEffect, useState } from 'react';
import { supabase } from '@/integrations/supabase/client';
import { useAuth } from '@/hooks/useAuth';

export interface PagePresenceUser {
  userId: string;
  timestamp: number;
}

export function usePagePresence(pageId: string | null) {
  const { user } = useAuth();
  const [activeUsers, setActiveUsers] = useState<PagePresenceUser[]>([]);

  useEffect(() => {
    if (!pageId || !user) return;

    const channel = supabase.channel(`page-presence-${pageId}`, {
      config: { presence: { key: user.id } },
    });

    channel
      .on('presence', { event: 'sync' }, () => {
        const state = channel.presenceState();
        const users: PagePresenceUser[] = [];
        for (const key of Object.keys(state)) {
          const presences = state[key] as Array<{ userId?: string; timestamp?: number }>;
          for (const p of presences) {
            if (p.userId) {
              users.push({ userId: p.userId, timestamp: p.timestamp ?? Date.now() });
            }
          }
        }
        // Deduplicate (mesma chave pode ter multiplos clients)
        const dedup = new Map<string, PagePresenceUser>();
        for (const u of users) {
          dedup.set(u.userId, u);
        }
        setActiveUsers(Array.from(dedup.values()));
      })
      .subscribe(async (status) => {
        if (status === 'SUBSCRIBED') {
          await channel.track({
            userId: user.id,
            timestamp: Date.now(),
          });
        }
      });

    return () => {
      supabase.removeChannel(channel);
      setActiveUsers([]);
    };
  }, [pageId, user]);

  return activeUsers;
}
```

```tsx
import React from 'react';
import { Avatar, AvatarFallback, AvatarImage } from '@/components/ui/avatar';
import { Tooltip, TooltipContent, TooltipProvider, TooltipTrigger } from '@/components/ui/tooltip';
import { useProfiles } from '@/hooks/useSupabaseData';
import { useAuth } from '@/hooks/useAuth';
import type { PagePresenceUser } from './usePagePresence';

interface Props {
  users: PagePresenceUser[];
  maxVisible?: number;
}

const getInitials = (name?: string | null) => {
  if (!name) return '?';
  return name.split(' ').map((n) => n[0]).join('').slice(0, 2).toUpperCase();
};

const PagePresenceIndicator: React.FC<Props> = ({ users, maxVisible = 3 }) => {
  const { data: profiles = [] } = useProfiles();
  const { user: me } = useAuth();
  // Excluir o usuario corrente (ele nao precisa se ver)
  const others = users.filter((u) => u.userId !== me?.id);
  if (others.length === 0) return null;

  const visible = others.slice(0, maxVisible);
  const overflow = others.length - visible.length;

  const getName = (userId: string) => {
    const p = profiles.find((pr: any) => pr.id === userId);
    return p?.name ?? 'Usuario';
  };

  const getAvatar = (userId: string) => {
    const p = profiles.find((pr: any) => pr.id === userId);
    return p?.avatar_url ?? null;
  };

  return (
    <TooltipProvider>
      <div className="flex items-center -space-x-2">
        {visible.map((u) => (
          <Tooltip key={u.userId}>
            <TooltipTrigger asChild>
              <Avatar className="h-6 w-6 border-2 border-background ring-1 ring-primary/40">
                <AvatarImage src={getAvatar(u.userId) ?? undefined} />
                <AvatarFallback className="text-[10px]">
                  {getInitials(getName(u.userId))}
                </AvatarFallback>
              </Avatar>
            </TooltipTrigger>
            <TooltipContent>
              <p className="text-xs">{getName(u.userId)} esta editando</p>
            </TooltipContent>
          </Tooltip>
        ))}
        {overflow > 0 && (
          <div className="h-6 w-6 rounded-full bg-muted text-[10px] flex items-center justify-center border-2 border-background">
            +{overflow}
          </div>
        )}
      </div>
    </TooltipProvider>
  );
};

export default PagePresenceIndicator;
```
