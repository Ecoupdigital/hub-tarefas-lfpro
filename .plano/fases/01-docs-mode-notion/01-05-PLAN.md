```typescript
export const useAllItemsForMention = (query: string) =>
  useQuery({
    queryKey: ['all-items-for-mention', query],
    staleTime: 30 * 1000,
    queryFn: async () => {
      const q = query.trim();
      let builder = supabase
        .from('items')
        .select('id, name, board_id, boards!inner(name)')
        .neq('state', 'deleted')
        .is('parent_item_id', null)
        .order('updated_at', { ascending: false })
        .limit(20);
      if (q.length >= 2) {
        builder = builder.ilike('name', `%${q}%`);
      }
      const { data, error } = await builder;
      if (error) throw error;
      return (data ?? []).map((row: any) => ({
        id: row.id as string,
        name: row.name as string,
        board_id: row.board_id as string,
        board_name: (row.boards?.name as string) ?? '',
      }));
    },
  });
```

```tsx
import React from 'react';
import { createReactInlineContentSpec } from '@blocknote/react';
import { useNavigate } from 'react-router-dom';
import { useQuery } from '@tanstack/react-query';
import { supabase } from '@/integrations/supabase/client';

interface MentionData {
  id: string;
  name: string;
  boardId: string;
  statusValue: { label: string; color: string } | null;
}

function useMentionData(itemId: string) {
  return useQuery<MentionData | null>({
    queryKey: ['mention-data', itemId],
    enabled: !!itemId,
    staleTime: 30 * 1000,
    queryFn: async () => {
      const { data: item, error } = await supabase
        .from('items').select('id, name, board_id').eq('id', itemId).single();
      if (error || !item) return null;
      const { data: statusCol } = await supabase
        .from('columns').select('id')
        .eq('board_id', item.board_id).eq('column_type', 'status')
        .order('position').limit(1).maybeSingle();
      let statusValue: { label: string; color: string } | null = null;
      if (statusCol) {
        const { data: cv } = await supabase
          .from('column_values').select('value')
          .eq('item_id', item.id).eq('column_id', statusCol.id).maybeSingle();
        if (cv?.value && typeof cv.value === 'object' && 'label' in (cv.value as object)) {
          statusValue = cv.value as { label: string; color: string };
        }
      }
      return { id: item.id, name: item.name, boardId: item.board_id, statusValue };
    },
  });
}

const MentionChip: React.FC<{ itemId: string }> = ({ itemId }) => {
  const navigate = useNavigate();
  const { data, isLoading } = useMentionData(itemId);
  if (isLoading) return <span className="inline-flex items-center gap-1 px-1.5 py-0.5 rounded bg-muted text-muted-foreground text-xs">@carregando...</span>;
  if (!data) return <span className="inline-flex items-center gap-1 px-1.5 py-0.5 rounded bg-destructive/10 text-destructive text-xs">@item indisponivel</span>;
  return (
    <span
      onClick={(e) => { e.preventDefault(); e.stopPropagation(); navigate(`/board/${data.boardId}?item=${data.id}`); }}
      role="link"
      className="inline-flex items-center gap-1 px-1.5 py-0.5 rounded cursor-pointer hover:bg-accent text-sm font-medium"
      style={{ backgroundColor: 'hsl(29 45% 71% / 0.18)', color: 'hsl(29 45% 30%)' }}
    >
      @{data.name}
      {data.statusValue && (
        <span className="ml-1 px-1 rounded text-xs font-normal" style={{ backgroundColor: data.statusValue.color, color: '#fff' }}>
          {data.statusValue.label}
        </span>
      )}
    </span>
  );
};

export const MentionInlineContent = createReactInlineContentSpec(
  { type: 'mention-item', propSchema: { itemId: { default: '' } }, content: 'none' },
  { render: (props) => <MentionChip itemId={props.inlineContent.props.itemId as string} /> }
);
```

```typescript
import { getDefaultReactSlashMenuItems, DefaultReactSuggestionItem } from '@blocknote/react';
import type { BlockNoteEditor } from '@blocknote/core';

interface SlashTriggerHandlers {
  onTriggerMention: () => void;
  onTriggerEmbedBoard?: () => void;
}

const TITLE_PT: Record<string, string> = {
  'Heading 1': 'Titulo 1',
  'Heading 2': 'Titulo 2',
  'Heading 3': 'Titulo 3',
  'Paragraph': 'Paragrafo',
  'Bullet List': 'Lista',
  'Numbered List': 'Lista numerada',
  'Check List': 'Checklist',
  'Code Block': 'Bloco de codigo',
  'Quote': 'Citacao',
  'Table': 'Tabela',
  'Image': 'Imagem',
  'Video': 'Video',
  'Audio': 'Audio',
  'File': 'Arquivo',
  'Emoji': 'Emoji',
};

const translate = (item: DefaultReactSuggestionItem): DefaultReactSuggestionItem => {
  const next = TITLE_PT[item.title];
  return next ? { ...item, title: next } : item;
};

export function getCustomSlashMenuItems(
  editor: BlockNoteEditor,
  handlers: SlashTriggerHandlers
): DefaultReactSuggestionItem[] {
  const defaults = getDefaultReactSlashMenuItems(editor).map(translate);
  const customs: DefaultReactSuggestionItem[] = [
    {
      title: 'Mencionar item',
      onItemClick: () => handlers.onTriggerMention(),
      aliases: ['mencionar', 'mention', '@', 'item'],
      group: 'LFPro',
      subtext: 'Inserir referencia a um item de board',
    },
  ];
  if (handlers.onTriggerEmbedBoard) {
    customs.push({
      title: 'Embedar board',
      onItemClick: () => handlers.onTriggerEmbedBoard!(),
      aliases: ['embedar', 'embed', 'board'],
      group: 'LFPro',
      subtext: 'Inserir tabela read-only de um board',
    });
  }
  return [...defaults, ...customs];
}
```
